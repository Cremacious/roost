```typescript
// apps/mobile/lib/auth.ts
// Expo auth client using better-auth expoClient plugin.
// Sessions are stored in expo-secure-store (iOS Keychain / Android Keystore).
// Import path used: [document the resolved path here]

import { createAuthClient } from "better-auth/react";
import * as SecureStore from "expo-secure-store";
import { expoClient } from "RESOLVED_IMPORT_PATH";

export const authClient = createAuthClient({
  baseURL: process.env.EXPO_PUBLIC_API_URL ?? "https://roost.vercel.app",
  plugins: [
    expoClient({
      scheme: "roostapp",
      storagePrefix: "roost",
      storage: SecureStore,
    }),
  ],
});

// Authenticated fetch — attaches session cookie to all API calls.
// Use this for every API call from Expo instead of plain fetch().
export async function authedFetch(url: string, init?: RequestInit) {
  const cookie = await authClient.getCookie();
  return fetch(url, {
    ...init,
    headers: {
      ...init?.headers,
      ...(cookie ? { cookie } : {}),
    },
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| apps/mobile → production API | All Expo API calls cross HTTPS to the Vercel production URL |
| expo-secure-store | Sessions stored in iOS Keychain / Android Keystore — encrypted at rest |
| trustedOrigins | Server rejects requests from schemes not in the allow-list |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-07 | Information Disclosure | Session storage (mobile) | mitigate | Use expo-secure-store (Keychain/Keystore) not AsyncStorage. `expoClient` handles this via the `storage: SecureStore` config. Never store session in AsyncStorage. |
| T-00-08 | Spoofing | trustedOrigins | mitigate | `"roostapp://"` added to trustedOrigins list in server auth config. Without this, the server rejects requests from the Expo app scheme. `"exp://**"` is for Expo Go dev only — acceptable for development. |
| T-00-09 | Information Disclosure | EXPO_PUBLIC_API_URL | accept | This variable is public (EXPO_PUBLIC_ prefix = included in bundle). It points to the production URL which is already public. No secret exposure. |
| T-00-10 | Tampering | better-auth Expo cookie bugs (issues #6810, #7674) | mitigate | AUTH-05 (Plan 06) is a live test against production Vercel URL specifically to catch residual bugs from issues #6810/#7674. The bug risk is accepted here and tested in Plan 06. |
</threat_model>
