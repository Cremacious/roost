## Task 2: `parseIngredients` helper

**Files:**
- Create: `src/lib/utils/parseIngredients.ts`

This helper deserializes both the old flat-string format (`["chicken", "garlic"]`) and the new structured format (`[{"name":"chicken","quantity":"2","unit":"cups"}]`). Old entries become `{ name: string }` objects. No migration of stored data needed.

- [ ] **Step 1: Create the helper**

```ts
// src/lib/utils/parseIngredients.ts

export type IngredientItem = {
  name: string
  quantity?: string
  unit?: string
}

export function parseIngredients(raw: string): IngredientItem[] {
  try {
    const parsed = JSON.parse(raw)
    if (!Array.isArray(parsed)) return []
    return parsed.map((item) => {
      if (typeof item === 'string') return { name: item }
      if (typeof item === 'object' && item !== null && typeof item.name === 'string') {
        return {
          name: item.name,
          quantity: item.quantity ?? undefined,
          unit: item.unit ?? undefined,
        }
      }
      return { name: String(item) }
    })
  } catch {
    return []
  }
}
```

```tsx
  ingredients,
  steps,
  onChange,
  expanded,
  onExpandedChange,
  showSteps = true,

    const next = ingredients.map((it, i) => i === idx ? { ...it, ...patch } : it)
    onChange(next, steps)

    onChange([...ingredients, { name: '' }], steps)

    onChange(ingredients.filter((_, i) => i !== idx), steps)
  }

  function updateStep(idx: number, value: string) {
    const next = steps.map((s, i) => i === idx ? value : s)
    onChange(ingredients, next)
  }

    onChange(ingredients, [...steps, ''])
  }

    onChange(ingredients, steps.filter((_, i) => i !== idx))

                  value={item.unit ?? ''}

                </select>
              </>
            )}
            <input
              style={INPUT_STYLE}
              placeholder="Ingredient name"
              value={item.name}
              onChange={e => updateIngredient(idx, { name: e.target.value })}
            />
            <button
              type="button"
              onClick={() => removeIngredient(idx)}
              style={{ width: 28, height: 28, borderRadius: 8, border: 'none', backgroundColor: 'transparent', color: '#EF4444', cursor: 'pointer', display: 'flex', alignItems: 'center', justifyContent: 'center', padding: 0 }}
```

## Task 4: `GroceryPushSheet` component

**Files:**
- Create: `src/components/meals/GroceryPushSheet.tsx`

Sheet that shows ingredient checkboxes (all checked by default) plus a list selector for premium users. Green "Add X ingredients" slab confirm button. Calls `POST /api/meals/[id]/add-to-grocery`.

- [ ] **Step 1: Create the component**

```tsx
// src/components/meals/GroceryPushSheet.tsx
'use client'

import { useState, useRef } from 'react'
import { useQuery } from '@tanstack/react-query'
import { toast } from 'sonner'
import { DraggableSheet } from '@/components/shared/DraggableSheet'
import type { IngredientItem } from '@/lib/utils/parseIngredients'

const GROCERY_COLOR = '#22C55E'
const GROCERY_DARK = '#159040'
const MEALS_COLOR = '#F97316'

interface GroceryList {
  id: string
  name: string
  isDefault: boolean
}

interface GroceryPushSheetProps {
  open: boolean
  onClose: () => void
  mealName: string
  ingredients: IngredientItem[]
  mealId: string
  isPremium: boolean
}

export function GroceryPushSheet({
  open, onClose, mealName, ingredients, mealId, isPremium,
}: GroceryPushSheetProps) {
  const [checked, setChecked] = useState<Record<number, boolean>>({})
  const [selectedListId, setSelectedListId] = useState<string | null>(null)
  const [busy, setBusy] = useState(false)

  const prevOpen = useRef(false)
  if (open && !prevOpen.current) {
    // Reset: check all by default
    const initial: Record<number, boolean> = {}
    ingredients.forEach((_, i) => { initial[i] = true })
    setChecked(initial)
    setSelectedListId(null)
  }
  prevOpen.current = open

  const { data: listsData } = useQuery({
    queryKey: ['grocery-lists'],
    queryFn: async () => {
      const r = await fetch('/api/grocery/lists')
      if (!r.ok) throw new Error('Failed')
      return r.json() as Promise<{ lists: GroceryList[] }>
    },
    enabled: open && isPremium,
    staleTime: 30_000,
  })

  const lists = listsData?.lists ?? []
  const defaultList = lists.find(l => l.isDefault)

  // Resolve the target list
  const targetListId = isPremium ? (selectedListId ?? defaultList?.id ?? null) : (defaultList?.id ?? null)
  const targetListName = lists.find(l => l.id === targetListId)?.name ?? 'Shopping List'

  const checkedCount = Object.values(checked).filter(Boolean).length

  async function handleConfirm() {
    if (checkedCount === 0) return
    setBusy(true)
    try {
      const ingredientNames = ingredients
        .filter((_, i) => checked[i])
        .map(it => it.quantity && it.unit
          ? `${it.quantity} ${it.unit} ${it.name}`
          : it.quantity
            ? `${it.quantity} ${it.name}`
            : it.name
        )

      const body: { listId?: string; ingredientNames: string[] } = { ingredientNames }
      if (targetListId) body.listId = targetListId

      const r = await fetch(`/api/meals/${mealId}/add-to-grocery`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      })
      if (!r.ok) throw new Error((await r.json()).error ?? 'Failed')
      toast.success(`Added ${checkedCount} ingredient${checkedCount === 1 ? '' : 's'} to ${targetListName}`)
      onClose()
    } catch (e) {
      toast.error('Could not add ingredients', { description: (e as Error).message })
    } finally {
      setBusy(false)
    }
  }

  return (
    <DraggableSheet open={open} onOpenChange={v => !v && onClose()} featureColor={MEALS_COLOR}>
      <div className="px-4 pb-8">
        <p className="mb-1 text-lg" style={{ color: 'var(--roost-text-primary)', fontWeight: 800 }}>
          {mealName}
        </p>
        <p style={{ fontSize: 13, fontWeight: 600, color: 'var(--roost-text-muted)', marginBottom: 16 }}>
          Choose ingredients to add to your grocery list.
        </p>

        {/* Ingredient checkboxes */}
        <div style={{ display: 'flex', flexDirection: 'column', gap: 8, marginBottom: 16 }}>
          {ingredients.map((item, idx) => {
            const isChecked = checked[idx] ?? true
            const displayName = item.quantity && item.unit
              ? `${item.quantity} ${item.unit} ${item.name}`
              : item.quantity
                ? `${item.quantity} ${item.name}`
                : item.name
            return (
              <button
                key={idx}
                type="button"
                onClick={() => setChecked(prev => ({ ...prev, [idx]: !prev[idx] }))}
                style={{
                  display: 'flex', alignItems: 'center', gap: 10, padding: '10px 12px',
                  borderRadius: 10, border: '1.5px solid var(--roost-border)',
                  borderBottom: '3px solid var(--roost-border)',
                  backgroundColor: isChecked ? `${GROCERY_COLOR}0F` : 'var(--roost-surface)',
                  cursor: 'pointer', textAlign: 'left',
                }}
              >
                <div style={{
                  width: 20, height: 20, borderRadius: 6, flexShrink: 0,
                  backgroundColor: isChecked ? GROCERY_COLOR : 'transparent',
                  border: `2px solid ${isChecked ? GROCERY_COLOR : 'var(--roost-border)'}`,
                  display: 'flex', alignItems: 'center', justifyContent: 'center',
                }}>
                  {isChecked && (
                    <svg width="11" height="8" viewBox="0 0 11 8" fill="none">
                      <path d="M1 4L4 7L10 1" stroke="white" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" />
                    </svg>
                  )}
                </div>
                <span style={{ fontSize: 14, fontWeight: 600, color: isChecked ? 'var(--roost-text-primary)' : 'var(--roost-text-muted)' }}>
                  {displayName}
                </span>
              </button>
            )
          })}
        </div>

        {/* List selector — premium only, shown when multiple lists exist */}
        {isPremium && lists.length > 1 && (
          <div style={{ marginBottom: 16 }}>
            <span style={{ fontSize: 11, fontWeight: 800, textTransform: 'uppercase', letterSpacing: '0.07em', color: '#374151', display: 'block', marginBottom: 8 }}>
              ADD TO LIST
            </span>
            <div style={{ display: 'flex', gap: 6, flexWrap: 'wrap' }}>
              {lists.map(list => {
                const active = (selectedListId ?? defaultList?.id) === list.id
                return (
                  <button
                    key={list.id}
                    type="button"
                    onClick={() => setSelectedListId(list.id)}
                    style={{
                      padding: '7px 14px', borderRadius: 10, fontSize: 13, fontWeight: 700,
                      border: 'none', borderBottom: `2px solid ${active ? GROCERY_DARK : 'var(--roost-border)'}`,
                      backgroundColor: active ? GROCERY_COLOR : 'var(--roost-surface)',
                      color: active ? '#fff' : 'var(--roost-text-secondary)', cursor: 'pointer',
                    }}
                  >
                    {list.name}
                  </button>
                )
              })}
            </div>
          </div>
        )}

        {/* Confirm button */}
        <button
          type="button"
          onClick={handleConfirm}
          disabled={busy || checkedCount === 0}
          style={{
            width: '100%', padding: '14px 0', borderRadius: 14, border: 'none',
            borderBottom: `3px solid ${GROCERY_DARK}`,
            backgroundColor: GROCERY_COLOR, color: '#fff', fontWeight: 800, fontSize: 15,
            cursor: busy || checkedCount === 0 ? 'not-allowed' : 'pointer',
            opacity: checkedCount === 0 ? 0.5 : busy ? 0.7 : 1,
          }}
        >
          {busy ? 'Adding...' : `Add ${checkedCount} ingredient${checkedCount === 1 ? '' : 's'}`}
        </button>
      </div>
    </DraggableSheet>
  )
}
```

```ts
// src/app/api/meals/[id]/add-to-grocery/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { getSession, getUserHousehold } from '@/lib/auth/helpers'
import { db } from '@/lib/db'
import { meals, groceryLists, groceryItems } from '@/db/schema'
import { eq, and, isNull } from 'drizzle-orm'
import { parseIngredients } from '@/lib/utils/parseIngredients'
import { logActivity } from '@/lib/utils/activity'

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  const { id } = await params
  const session = await getSession()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const membership = await getUserHousehold(session.user.id)
  if (!membership) return NextResponse.json({ error: 'No household' }, { status: 403 })

  const { householdId } = membership

  // Verify meal belongs to household
  const [meal] = await db
    .select()
    .from(meals)
    .where(and(eq(meals.id, id), eq(meals.householdId, householdId), isNull(meals.deletedAt)))
    .limit(1)

  if (!meal) return NextResponse.json({ error: 'Meal not found' }, { status: 404 })

  const body = await req.json().catch(() => ({}))
  const { listId, ingredientNames } = body as { listId?: string; ingredientNames?: string[] }

  // Resolve list — use provided listId or fall back to default list
  let resolvedListId = listId
  if (!resolvedListId) {
    const [defaultList] = await db
      .select({ id: groceryLists.id })
      .from(groceryLists)
      .where(
        and(
          eq(groceryLists.householdId, householdId),
          eq(groceryLists.isDefault, true),
          isNull(groceryLists.deletedAt)
        )
      )
      .limit(1)
    resolvedListId = defaultList?.id
  }

  if (!resolvedListId) {
    return NextResponse.json({ error: 'No grocery list found' }, { status: 400 })
  }

  // Resolve ingredient names — use provided list or all meal ingredients
  let names: string[]
  if (ingredientNames && ingredientNames.length > 0) {
    names = ingredientNames
  } else {
    names = parseIngredients(meal.ingredients).map(it =>
      it.quantity && it.unit
        ? `${it.quantity} ${it.unit} ${it.name}`
        : it.quantity
          ? `${it.quantity} ${it.name}`
          : it.name
    )
  }

  if (names.length === 0) {
    return NextResponse.json({ error: 'No ingredients to add' }, { status: 400 })
  }

  // Insert grocery items
  await db.insert(groceryItems).values(
    names.map(name => ({
      listId: resolvedListId!,
      name,
      addedBy: session.user.id,
    }))
  )

  await logActivity({
    householdId,
    userId: session.user.id,
    type: 'item_added',
    entityId: id,
    entityType: 'meal',
    description: `Added ${names.length} ingredient${names.length === 1 ? '' : 's'} from ${meal.name} to grocery list`,
  })

  return NextResponse.json({ ok: true, count: names.length })
}
```

```ts
export async function POST(req: NextRequest) {
  const session = await getSession()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const membership = await getUserHousehold(session.user.id)
  if (!membership) return NextResponse.json({ error: 'No household' }, { status: 403 })

  const { householdId } = membership
  const body = await req.json()
  const { mealId: providedMealId, name, slotDate, slotType, savedToBank } = body

  if (!slotDate || !slotType) {
    return NextResponse.json({ error: 'slotDate and slotType are required' }, { status: 400 })
  }

  let resolvedMealId = providedMealId

  // If name provided instead of mealId, create the meal inline
  if (!resolvedMealId && name) {
    const [created] = await db
      .insert(meals)
      .values({
        householdId,
        name: name.trim(),
        ingredients: '[]',
        savedToBank: savedToBank === true,  // default false for quick-add
        createdBy: session.user.id,
      })
      .returning({ id: meals.id })
    resolvedMealId = created.id
  }

  if (!resolvedMealId) {
    return NextResponse.json({ error: 'mealId or name is required' }, { status: 400 })
  }

  // Delete existing slot for same day+type (upsert behavior)
  const existing = await db
    .select({ id: mealPlanSlots.id })
    .from(mealPlanSlots)
    .where(
      and(
        eq(mealPlanSlots.householdId, householdId),
        eq(mealPlanSlots.slotDate, slotDate),
        eq(mealPlanSlots.slotType, slotType)
      )
    )
    .limit(1)

  if (existing.length > 0) {
    await db.delete(mealPlanSlots).where(eq(mealPlanSlots.id, existing[0].id))
  }

  const [slot] = await db
    .insert(mealPlanSlots)
    .values({
      householdId,
      mealId: resolvedMealId,
      slotDate,
      slotType,
      createdBy: session.user.id,
    })
    .returning()

  return NextResponse.json({ slot }, { status: 201 })
}
```
